$arquivos = @(
  "src\app\app.component.ts",
  "src\app\shared\components\child-selector\child-selector.component.ts",
  "src\app\shared\components\vaccine-card\vaccine-card.component.ts",
  "src\app\shared\components\campaign-banner\campaign-banner.component.ts",
  "src\app\features\dashboard\dashboard.page.ts",
  "src\app\features\vaccine-history\vaccine-history.page.ts",
  "src\app\features\child-profile\child-profile.page.ts",
  "src\app\features\campaigns\campaigns.page.ts"
)

foreach ($arquivo in $arquivos) {
  $caminho = Join-Path $PSScriptRoot $arquivo
  if (Test-Path $arquivo) {
    $conteudo = Get-Content $arquivo -Raw
    if ($conteudo -notmatch "standalone") {
      $conteudo = $conteudo -replace "@Component\(\{", "@Component({`n  standalone: false,"
      Set-Content $arquivo $conteudo -NoNewline
      Write-Host "standalone: false adicionado em $arquivo" -ForegroundColor Green
    } else {
      Write-Host "standalone ja existe em $arquivo" -ForegroundColor Yellow
    }
  } else {
    Write-Host "Arquivo nao encontrado: $arquivo" -ForegroundColor Red
  }
}

Write-Host "`nConcluido! Rode 'ionic serve' para testar." -ForegroundColor Cyan
